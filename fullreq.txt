<div class="outerpair1"><div class="outerpair2"><div class="shadowbox"><div class="innerbox"><h1>Hjælp til søgning</h1>
<p>
<strong>Adresse</strong>
</p>
<p>Søg på et område (fx 'Fyn'), en by (fx 'Vejle'), et eller flere postnumre (fx '6000' eller '2210-8230' eller '2100, 2200, 2300').<br/>Du kan også søge på ejendommens navn eller et vejnavn.</p>
<p>
<strong>Boligønsker</strong>
</p>
<p>Indtast dine ønsker for minimum- og maksimum husleje, antal værelser, boligareal og etage - eller lad felterne være tomme.</p>
<p>
<b>Boligtype og Faciliteter</b>
</p>
<p>Her kan du angive mere detaljerede ønsker til din bolig.<br/>Giver din søgning for mange boliger, kan du markere flere ønsker her. Giver din søgning for få boliger, kan du prøve at fjerne dine markeringer.</p>
<p>

    function getValueFromQueryString(key) {
        key = key.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
        var regex = new RegExp("[\\?&]" + key + "=([^&#]*)");
        var qs = regex.exec(window.location.href);

        if (qs == null)
            return "";
        else
            return qs[1];
    } 

    
    function doSearchOnList()
    {
        var sa = getValueFromQueryString('sa');
        var url = UrlListResult + "?" + getQueryString().queryString;

        if (sa != '') {
            // Add search agent id to url if it already was in the url (edit mode).
            url = url + '&sa=' + sa;
        }

        if (getValueFromQueryString('showdialog') == 1) {
            // Remove showdialog value if found.
            url = removeValueFromUrl(url, 'showdialog');
            // Add showdialog with new value (keep dialog open).
            url = url + "&showdialog=2";
        } else if (getValueFromQueryString('showdialog') == 2) {
            url = url + "&showdialog=2";
        }

        document.location.href = url;
    }
